# docker

# 基本命令

-- 查看docker版本
docker version

-- 查看当前所有的镜像
docker images

-- 删除某个镜像
docker rmi 镜像id

-- 查看当前所有的容器(包括停止的)
docker ps -a

-- 查看当前运行中的容器
docker ps

-- 停止某个容器的运行
docker stop 容器id

-- 删除某个容器(需要先停止该容器的运行才可删除)
docker rm 容器id

-- 查看指定容器的ip地址（打印出来的是桥接方式的ip，对应网络docker0）
docker inspect --format '{{ .NetworkSettings.IPAddress }}' 容器id

# 运行相关

# 1.创建以某个镜像为基础的容器
docker run --privileged=true --name openresty-demo1 -p 8080:8080 -p 8081:8081 -p 8082:8082 -p 8083:8083 -v /media/ryan/disk11/docker-nginx/logs:/usr/local/openresty/nginx/logs -v /media/ryan/disk11/docker-nginx/conf/conf.d:/etc/nginx/conf.d -v /media/ryan/disk11/docker-nginx/conf/nginx.conf:/usr/local/openresty/nginx/conf/nginx.conf -d openresty/openresty
  • --privileged=true: 容器内的root拥有真正root权限，否则容器内root只是外部普通用户权限
  • --name: 后面指定该容器的名称，如果所指定的容器名称已存在，将创建失败
  • -p: 后面指定端口号映射关系，冒号左边为宿主机的端口，冒号右边为容器中的端口
  • -v: 后面指定挂载的volumn(卷)或文件，冒号左边为宿主机的目录/文件，冒号右边为容器中的目录/文件，说白了就是数据外挂在宿主机上；每一个-v后只能跟一对volumn映射
  • -d: 表示此容器将运行在后台(守护运行)
  • openresty/openresty: 这个是镜像名称，也可以是镜像id
# 2.进入容器中
-- 这种方式进入容器中执行exit命令并不会导致容器停止运行
docker exec -it 容器id /bin/bash
# 3.容器和宿主机之间的文件互传
-- 3.1 复制容器文件到宿主机  如将容器中的/home/test.md文件复制到宿主机的/home/tmp目录下
docker cp 容器id:/home/test.md /home/tmp

-- 3.2 复制宿主机文件到容器  如将宿主机的/home/hiahia.md文件复制到容器中的/home/con目录下
docker cp /home/hiahia.md 容器id:/home/con
